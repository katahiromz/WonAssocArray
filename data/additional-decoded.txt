// @implemented
HRESULT CEnumAssocElems::Next(
    ULONG celt,
    IAssociationElementOld *pElement,
    ULONG *pceltFetched)
{
    ULONG cGot = 0;
    ZeroMemory(pElement, 4 * celt);
    if (celt)
    {
        do
        {
            if (!_Next(pElement))
                break;
            ++m_iItem;
            ++cGot;
            ++pElement;
        } while (cGot < celt);
    }

    if (pceltFetched)
        *pceltFetched = cGot;
    return (celt == cGot) ? S_OK : S_FALSE;
}
